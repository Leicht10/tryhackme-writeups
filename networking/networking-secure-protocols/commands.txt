# Networking Secure Protocols - Quick Commands

# SSH
ssh user@192.168.1.50
ssh -p 2222 user@host

# Simple OpenSSL client test (connect to TLS service)
openssl s_client -connect example.com:443 -servername example.com

# Show TLS certificate chain
openssl s_client -connect example.com:443 -showcerts

# Generate RSA keypair (example)
openssl genpkey -algorithm RSA -out key.pem -pkeyopt rsa_keygen_bits:2048
openssl req -new -key key.pem -out csr.pem

# Test HTTPS with curl
curl -v https://example.com
curl -Iv https://example.com

# Start a local SSH tunnel (port forwarding)
ssh -L 8888:internal.host:80 user@bastion.host
